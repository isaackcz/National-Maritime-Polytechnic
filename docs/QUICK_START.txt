     → Called when you click "Activity History" tab
     → Shows your recent actions

📝 CODE EXAMPLE - API CALL WITH COMMENTS:
━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━━

  /**
   * API CALL: GET - Fetch trainee personal information from database
   * This gets all saved personal info including contact, education, etc.
   * Endpoint: /my-account/get_trainee_general_info
   */
  const fetchPersonalInformation = async () => {
      try {
          // Get authentication token from browser storage
          const token = getToken("csrf-token");
          
          // API CALL: GET request to get trainee's complete information
          const response = await axios.get(
              `${url}/my-account/get_trainee_general_info`, 
              {
                  headers: {
                      Authorization: `Bearer ${token}`, // For authentication
                  }
              }
          );

          // If successful, fill form with data
          if(response.status === 200) {
              const data = response.data.trainee_general_info?.[0];
              setFirstName(data.fname || '');
              setLastName(data.lname || '');
              // ... etc
          }
      } catch (error) {
          console.error("Error:", error);
          alert('Failed to load data');
      }
  };
